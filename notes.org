# Build a query to get all the pull requests and their most recent activity.
query {
  organization(login:"github") {
    repositories(first: 100) {
      nodes {
        pullRequests(states:[OPEN], first:100) {
					nodes {
            author {
              login
            },
            title,
            url,
            publishedAt,
            number,
            lastEditedAt,
            timeline(last: 1) {

            }
          }
        }
      }
    }
  }
}
